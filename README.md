# build-liboqs
```
cmake \
-GNinja \
-S "." \
-B "build" \
-DCMAKE_BUILD_TYPE='Release' \
-DCMAKE_VERBOSE_MAKEFILE:BOOL=ON \
-DBUILD_SHARED_LIBS:BOOL=ON \
-DCMAKE_INSTALL_SO_NO_EXE:INTERNAL=0 \
-DOQS_ENABLE_KEM_BIKE:BOOL=ON \
-DOQS_ENABLE_KEM_FRODOKEM:BOOL=ON \
-DOQS_ENABLE_KEM_HQC:BOOL=ON \
-DOQS_ENABLE_KEM_KYBER:BOOL=OFF \
-DOQS_ENABLE_KEM_ML_KEM:BOOL=ON \
-DOQS_ENABLE_KEM_NTRUPRIME:BOOL=ON \
-DCMAKE_INSTALL_PREFIX:PATH=/usr \
-DINCLUDE_INSTALL_DIR:PATH=/usr/include \
-DSYSCONF_INSTALL_DIR:PATH=/etc \
-DSHARE_INSTALL_PREFIX:PATH=/usr/share \
-DLIB_INSTALL_DIR:PATH=/usr/lib/x86_64-linux-gnu
ninja -C "build" -j $(nproc --all) --verbose
DESTDIR="/tmp/liboqs" cmake --install "build"
```
